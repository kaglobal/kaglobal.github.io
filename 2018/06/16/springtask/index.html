Spring task 定时任务

2018-06-16
java技术-springboot
kabai

配置文件:
xmlns:task=”http://www.springframework.org/schema/task"
http://www.springframework.org/schema/task/spring-task-4.1.xsd

添加实现类所在包注解:
<context:component-scan base-package=”com.hl.mtn.util” />

<task:annotation-driven scheduler="qbScheduler" mode="proxy"/>
<task:scheduler id="qbScheduler" pool-size="5"/>

实现类:
@Component(“AlterIotmetersJob”)
public class AlterIotmetersJob {

@Autowired
private  IotMeterInfoServices iotMeterInfoServices;

private Logger logger = Logger.getLogger(AlterIotmetersJob.class);

 @Scheduled(cron = "0 0 23 * * ?")
    public void job1() {
     int Publicdd= Integer.parseInt(PublicConfig.turnMonitor);
     if(Publicdd==0) {

        logger.info("执行定时删除历史数据:");
        iotMeterInfoServices.deleteMeterDatasByTime();
        iotMeterInfoServices.deleteMeterShipByTime();
     }

    }

}
